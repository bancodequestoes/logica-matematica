\question[10]

Encontre as variáveis livres nas seguintes formulas:

\begin{parts}
	\part $p(x) \land \neg r(y, a)$ \fillin[x e y livres]
    \part $\exists x(r(x, y))$ \fillin[y livre]
	\part $\forall x(p(x)) \rightarrow \exists y(\neg q(f(x), y, f(y))) $ \fillin[x livre]
	\part $\forall x \exists y(r(x,f(y)))$ \fillin[Não há variáveis livres]
\end{parts}
