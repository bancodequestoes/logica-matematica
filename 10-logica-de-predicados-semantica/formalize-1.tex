\question[10]

Formalize as seguintes sequências:

\begin{parts}
	\part \textit{``Todos os estudantes são inteligentes'' } \newline \fillin[$\forall (estudante(x) \rightarrow inteligente(x))$][15cm]
    
    \part \textit{``Existe um estudante''} \newline \fillin[$\exists x(estudante(x))$][15cm]
    
    \part \textit{``Existe um estudante inteligente''} \newline \fillin[$\exists x(estudante(x)  \wedge inteligente(x))$][15cm]
    
    \part \textit{``Todo estudante ama algum estudante''} \newline \fillin[$\forall x(estudante(x) \rightarrow \exists y(estudante(y) \wedge ama(x, y)))$][15cm]
\end{parts}
