\question[10]

Encontre as variáveis livres nas seguintes fórmulas:

\begin{parts}
	\part $p(x) \wedge \neg r(y, a)$ \fillin[x e y são livres]
    \part $\exists x(r(x, y))$ \fillin[y]
	\part $\forall x(\exists y  (r(x,f(y))) \rightarrow r(x,y)) $ \fillin[y em r(x,y) é livre]
	\part $\forall x(p(x) \rightarrow \exists y(\neg q(f(x), y, f(y))))$ \fillin[Não há variáveis livres]
\end{parts}
