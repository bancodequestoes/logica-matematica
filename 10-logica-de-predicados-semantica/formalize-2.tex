\question[10]

Formalize as seguintes sequências:

\begin{parts}
	\part \textit{``Pedro é um estudante'' } \newline \fillin[$estudante(Pedro)$][15cm]
    
    \part \textit{``Pedro estuda Lógica e Geometria''} \newline \fillin[$estuda(Pedro, Logica) \wedge estuda(Pedro, Geometria)$][15cm]
    
    \part \textit{``Pedro não estuda Lógica''} \newline \fillin[$\neg estuda(Pedro, Logica)$][15cm]
    
    \part \textit{``Nenhum estudante ama Prova''} \newline \fillin[$\neg \exists x (estudante(x) \wedge ama (x, Prova)))$][15cm]
\end{parts}
